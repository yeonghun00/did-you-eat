rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to access their own user document (for FCM tokens)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Allow access to families collection for authenticated users with enhanced security
    match /families/{familyId} {
      // Basic read/write for authenticated users
      allow read, write: if request.auth != null;

      // More granular rules for specific operations
      allow read: if request.auth != null;
      allow write: if request.auth != null;

      // Enhanced rules for meals subcollection
      match /meals/{date} {
        allow read: if request.auth != null;
        allow write: if request.auth != null;
      }

      // Rules for analytics and monitoring subcollections
      match /analytics/{document=**} {
        allow read: if request.auth != null;
        allow write: if request.auth != null;
      }

      // ðŸ†• ADD THIS: Rules for child_devices subcollection (for FCM tokens)
      match /child_devices/{deviceId} {
        allow read, write: if request.auth != null;
      }
    }
  }
}
